35void ToggleButtonInput(ToggleButton &toggle_button, SDL_Event &e) {
36 SDL_DisplayMode mode;
37 SDL_GetDisplayMode(0, 0, &mode);
38 int window_width = mode.w;
39 int window_height = mode.h;
40 SDL_Point touch_point = {
41 static_cast<int>((float)e.tfinger.x * (float)window_width),
42 static_cast<int>((float)e.tfinger.y * (float)window_height)
43 };
44 SDL_Rect _rect = {toggle_button.rect.x, toggle_button.rect.y, toggle_button.rect.w - toggle_button.rect.x, toggle_button.rect.h - toggle_button.rect.y};
45
46 switch (e.type) {
47 case SDL_FINGERDOWN:
48 if (!toggle_button.active && SDL_PointInRect(&touch_point, &_rect)) {
49 toggle_button.FD = true;
50 break;
51 }
52 if (toggle_button.active && SDL_PointInRect(&touch_point, &_rect)) {
53 toggle_button.FD = true;
54 break;
55 }
56
57 case SDL_FINGERUP:
58 if (!toggle_button.active && SDL_PointInRect(&touch_point, &_rect) && toggle_button.FD) {
59 toggle_button.active = true;
60 toggle_button.FD = false;
61 break;
62 }
63 if (toggle_button.active && SDL_PointInRect(&touch_point, &_rect) && toggle_button.FD) {
64 toggle_button.active = false;
65 toggle_button.FD = false;
66 break;
67 }
68 toggle_button.FD = false;
69 }
70}
