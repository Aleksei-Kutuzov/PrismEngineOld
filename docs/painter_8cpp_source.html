8void PRISM_Painter::DrawThickLine(SDL_Renderer* renderer, int x1, int y1, int x2, int y2, int thickness) {
9 // Вычисление разности координат
10 int dx = x2 - x1;
11 int dy = y2 - y1;
12
13 // Вычисление нормализованного вектора перпендикулярного линии
14 float length = (float)sqrtf(dx * dx + dy * dy);
15 float nx = -dy / length;
16 float ny = dx / length;
17
18 // Рисование параллельных линий
19 for (int i = -thickness / 2; i <= thickness / 2; ++i) {
20 int offsetX = (int)(nx * i);
21 int offsetY = (int)(ny * i);
22 SDL_RenderDrawLine(renderer, x1 + offsetX, y1 + offsetY, x2 + offsetX, y2 + offsetY);
23 }
24}
