30{
31 PRISM_Vector3d v;

38PRISM_Vector3d Math::Vector_IntersectPlane(PRISM_Vector3d& plane_p, PRISM_Vector3d& plane_n, PRISM_Vector3d& lineStart, PRISM_Vector3d& lineEnd)
39{
40 plane_n = Vector_Normalise(plane_n);
41 float plane_d = -Vector_DotProduct(plane_n, plane_p);
42 float ad = Vector_DotProduct(lineStart, plane_n);
43 float bd = Vector_DotProduct(lineEnd, plane_n);
44 float t = (-plane_d - ad) / (bd - ad);
45 PRISM_Vector3d lineStartToEnd = lineEnd - lineStart;
46 PRISM_Vector3d lineToIntersect = lineStartToEnd * t;
47 return lineStart + lineToIntersect;
48}

119PRISM_Matrix_4X4 Math::Matrix_MakeProjection(float fFovDegrees, float fAspectRatio, float fNear, float fFar)
120{
121 // float fFovRad = 1.0f / tanf(fFovDegrees * 0.5f / 180.0f * 3.14159f);
122 float fFovRad = fFovDegrees;
123 PRISM_Matrix_4X4 matrix;
124 matrix.m[0][0] = fAspectRatio * fFovRad;
125 matrix.m[1][1] = fFovRad;
126 matrix.m[2][2] = fFar / (fFar - fNear);
127 matrix.m[3][2] = (-fFar * fNear) / (fFar - fNear);
128 matrix.m[2][3] = 1.0f;
129 matrix.m[3][3] = 0.0f;
130 return matrix;
131}

142PRISM_Matrix_4X4 Math::Matrix_PointAt(PRISM_Vector3d& pos, PRISM_Vector3d& target, PRISM_Vector3d& up)
143{
144 // Calculate new forward direction
145 PRISM_Vector3d newForward = target - pos;
146 newForward = Math::Vector_Normalise(newForward);
147
148 // Calculate new Up direction
149 PRISM_Vector3d a = newForward * Math::Vector_DotProduct(up, newForward);
150 PRISM_Vector3d newUp = up - a;
151 newUp = Math::Vector_Normalise(newUp);
152
153 // New Right direction is easy, its just cross product
154 PRISM_Vector3d newRight = Math::Vector_CrossProduct(newUp, newForward);
155
156 // Construct Dimensioning and Translation Matrix
157 PRISM_Matrix_4X4 matrix;
158 matrix.m[0][0] = newRight.x; matrix.m[0][1] = newRight.y; matrix.m[0][2] = newRight.z; matrix.m[0][3] = 0.0f;
159 matrix.m[1][0] = newUp.x; matrix.m[1][1] = newUp.y; matrix.m[1][2] = newUp.z; matrix.m[1][3] = 0.0f;
160 matrix.m[2][0] = newForward.x; matrix.m[2][1] = newForward.y; matrix.m[2][2] = newForward.z; matrix.m[2][3] = 0.0f;
161 matrix.m[3][0] = pos.x; matrix.m[3][1] = pos.y; matrix.m[3][2] = pos.z; matrix.m[3][3] = 1.0f;
162 return matrix;
163}
