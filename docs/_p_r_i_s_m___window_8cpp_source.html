6PRISM_Window::PRISM_Window(int width, int height, const char* title)
7 : window(nullptr), glContext(nullptr), running(true)
8{
9 this->w = width;
10 this->h = height;
11 // Инициализация SDL
12 if (SDL_Init(SDL_INIT_VIDEO) < 0) {
13 SDL_LogError(SDL_LOG_PRIORITY_ERROR, "SDL could not initialize! SDL_Error: %s", SDL_GetError());
14 running = false;
15 return;
16 }
17
18 // Установка кодировки UTF-8
19 if (!SDL_SetHint(SDL_HINT_WINDOWS_ENABLE_MESSAGELOOP, "1") ||
20 !SDL_SetHint(SDL_HINT_VIDEO_WIN_D3DCOMPILER, "d3dcompiler_47.dll") ||
21 !SDL_SetHint(SDL_HINT_VIDEO_ALLOW_SCREENSAVER, "1") ||
22 !SDL_SetHint(SDL_HINT_VIDEO_X11_NET_WM_BYPASS_COMPOSITOR, "0") ||
23 !SDL_SetHint(SDL_HINT_VIDEO_X11_XRANDR, "1")) {
24 SDL_LogWarn(SDL_LOG_PRIORITY_WARN, "Failed to set SDL hints!");
25 }
26
27 SDL_GL_SetAttribute(SDL_GL_MULTISAMPLEBUFFERS, 1);
28 SDL_GL_SetAttribute(SDL_GL_MULTISAMPLESAMPLES, 4); // Количество сэмплов (4x MSAA)
29
30
31 SDL_LogInfo(SDL_LOG_PRIORITY_INFO, "SDL initialized successfully.");
32
33 // Создание окна с поддержкой OpenGL
34 window = SDL_CreateWindow(title,
35 SDL_WINDOWPOS_UNDEFINED, SDL_WINDOWPOS_UNDEFINED,
36 width, height, SDL_WINDOW_OPENGL | SDL_WINDOW_RESIZABLE);
37
38 if (!window) {
39 SDL_LogError(SDL_LOG_PRIORITY_ERROR, "Window could not be created! SDL_Error: %s", SDL_GetError());
40 running = false;
41 return;
42 }
43
44 // Создание контекста OpenGL
45 glContext = SDL_GL_CreateContext(window);
46 if (!glContext) {
47 SDL_LogError(SDL_LOG_PRIORITY_ERROR, "OpenGL context could not be created! SDL_Error: %s", SDL_GetError());
48 running = false;
49 return;
50 }
51
52 // Инициализация OpenGL через freeGLUT (необязательно, но полезно для расширенных возможностей)
53 int argc = 0;
54 glutInit(&argc, nullptr);
55
56 // Настройка OpenGL (например, для использования глубины)
57 glEnable(GL_DEPTH_TEST);
58 glDepthFunc(GL_LESS);
59
60
61 SDL_LogInfo(SDL_LOG_PRIORITY_INFO, "Window created successfully, OpenGL context initialized.");
62 //glEnable(GL_MULTISAMPLE); не все версии поддерживают версиях
63
64 // Настройка ортографической проекции для координат SDL
65 glMatrixMode(GL_PROJECTION);
66 glLoadIdentity();
67 glOrtho(0, width, height, 0, -1, 1);
68
69 // Сброс модели
70 glMatrixMode(GL_MODELVIEW);
71 glLoadIdentity();
72
73 // Устанавливаем единичную модельно-видовую матрицу
74 glMatrixMode(GL_MODELVIEW);
75 glLoadIdentity();
76
77}

100void PRISM_Window::handleEvents() {
101 SDL_Event e;
102 while (SDL_PollEvent(&e)) {
103 if (e.type == SDL_QUIT) {
104 running = false;
105 }
106 else if (e.type == SDL_WINDOWEVENT && e.window.event == SDL_WINDOWEVENT_RESIZED) {
107 int newWidth = e.window.data1;
108 int newHeight = e.window.data2;
109
110 // Обновляем видовой порт и проекцию
111 glViewport(0, 0, newWidth, newHeight);
112
113 glMatrixMode(GL_PROJECTION);
114 glLoadIdentity();
115 glOrtho(0, newWidth, newHeight, 0, -1, 1);
116 glMatrixMode(GL_MODELVIEW);
117 glLoadIdentity();
118 w = newWidth; h = newHeight;
119 }
120 }
121}
