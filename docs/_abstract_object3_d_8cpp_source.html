100PRISM_Color AbstractObject3D::CalculatePhongShadingColor(const PRISM_Vector3d& normal, const PRISM_Vector3d& viewDir,
101 const PRISM_Light& light, const PRISM_Vector3d& fragPos) {
102 // Ambient component
103 float ambientStrength = 0.1f;
104 PRISM_Color ambient = { static_cast<Uint8>(light.color.r * ambientStrength),
105 static_cast<Uint8>(light.color.g * ambientStrength),
106 static_cast<Uint8>(light.color.b * ambientStrength),
107 light.color.a };
108
109 // Light direction (from fragment position to light position)
110 PRISM_Vector3d lightDir = { light.position.x - fragPos.x, light.position.y - fragPos.y, light.position.z - fragPos.z };
111 float lightLen = sqrt(lightDir.x * lightDir.x + lightDir.y * lightDir.y + lightDir.z * lightDir.z);
112 lightDir = { lightDir.x / lightLen, lightDir.y / lightLen, lightDir.z / lightLen };
113
114 // Diffuse component
115 float diff = std::max(0.0f, lightDir.x * normal.x + lightDir.y * normal.y + lightDir.z * normal.z);
116 PRISM_Color diffuse = { static_cast<Uint8>(light.color.r * diff),
117 static_cast<Uint8>(light.color.g * diff),
118 static_cast<Uint8>(light.color.b * diff),
119 light.color.a };
120
121 // Reflect direction
122 PRISM_Vector3d reflectDir = { lightDir.x - 2 * normal.x * diff,
123 lightDir.y - 2 * normal.y * diff,
124 lightDir.z - 2 * normal.z * diff };
125 float spec = std::pow(std::max(0.0f, reflectDir.x * viewDir.x + reflectDir.y * viewDir.y + reflectDir.z * viewDir.z), 32);
126 PRISM_Color specular = { static_cast<Uint8>(light.color.r * spec),
127 static_cast<Uint8>(light.color.g * spec),
128 static_cast<Uint8>(light.color.b * spec),
129 light.color.a };
130
131 // Final color
132 PRISM_Color result = { static_cast<Uint8>(std::min(255, ambient.r + diffuse.r + specular.r)),
133 static_cast<Uint8>(std::min(255, ambient.g + diffuse.g + specular.g)),
134 static_cast<Uint8>(std::min(255, ambient.b + diffuse.b + specular.b)),
135 light.color.a };
136 return result;
137}

290// Функция для вычисления барицентрических координат
291bool AbstractObject3D::ComputeBarycentricCoords(float x, float y, const PRISM_Vector3d& v0, const PRISM_Vector3d& v1, const PRISM_Vector3d& v2, float& u, float& v, float& w) {
292 float denom = (v1.y - v2.y)*(v0.x - v2.x) + (v2.x - v1.x)*(v0.y - v2.y);
293 u = ((v1.y - v2.y)*(x - v2.x) + (v2.x - v1.x)*(y - v2.y)) / denom;
294 v = ((v2.y - v0.y)*(x - v2.x) + (v0.x - v2.x)*(y - v2.y)) / denom;
295 w = 1.0f - u - v;
296 return u >= 0 && v >= 0 && w >= 0;
297}

356void AbstractObject3D::OptimizedEdgeRasterization(int x1, int y1, int x2, int y2, int x3, int y3, float z1, float z2, float z3,
357 SDL_Renderer* SDL_renderer, PRISM_Light light, short rgba[4], PRISM_RenderMode RendMode) {
358 PRISM_Vector3d v0 = {static_cast<float>(x1), static_cast<float>(y1), static_cast<float>(z1)};
359 PRISM_Vector3d v1 = {static_cast<float>(x2), static_cast<float>(y2), static_cast<float>(z2)};
360 PRISM_Vector3d v2 = {static_cast<float>(x3), static_cast<float>(y3), static_cast<float>(z3)};
361 // Сортировка вершин по y (v0.y <= v1.y <= v2.y)
362 if (v1.y < v0.y) std::swap(v0, v1);
363 if (v2.y < v0.y) std::swap(v0, v2);
364 if (v2.y < v1.y) std::swap(v1, v2);
365 // Функция для интерполяции X-координаты по рёбрам треугольника
366 auto interpolateX = [](float x0, float y0, float x1, float y1, float y) -> float {
367 if (y0 == y1) return x0; // Избегаем деления на ноль
368 return x0 + (x1 - x0) * (y - y0) / (y1 - y0);
369 };
370
371 // Инкрементальная растеризация: делим треугольник на два части (верхняя и нижняя)
372 auto rasterize_half = [&](PRISM_Vector3d v0, PRISM_Vector3d v1, PRISM_Vector3d v2) {
373 for (int y = static_cast<int>(std::ceil(v0.y)); y <= static_cast<int>(std::floor(v2.y)); ++y) {
374 float xStart = interpolateX(v0.x, v0.y, v2.x, v2.y, y);
375 float xEnd = interpolateX(v1.x, v1.y, v2.x, v2.y, y);
376 if (xStart > xEnd) std::swap(xStart, xEnd);
377 for (int x = static_cast<int>(std::ceil(xStart)); x <= static_cast<int>(std::floor(xEnd)); ++x) {
378 float u, v, w;
379 if (ComputeBarycentricCoords(static_cast<float>(x), static_cast<float>(y), v0, v1, v2, u, v, w)) {
380 float z = u * v0.z + v * v1.z + w * v2.z;
381 if (z < Camera.z_buffer.get(x, y) || !RendMode.Z_Buffer) {
382 Uint8 r = rgba[0];
383 Uint8 g = rgba[1];
384 Uint8 b = rgba[2];
385 Uint8 a = rgba[3];
386
387 SDL_SetRenderDrawColor(SDL_renderer, r, g, b, a);
388 SDL_RenderDrawPoint(SDL_renderer, x, y);
389 Camera.z_buffer.update(x, y, z); // Обновляем Z-буфер
390 }
391 }
392 }
393 }
394 };
395 // Растеризуем верхний и нижний треугольники
396 rasterize_half(v0, v1, v2); // Нижняя часть треугольника
397 rasterize_half(v1, v1, v2); // Верхняя часть
398}

453 SDL_Renderer* SDL_renderer, PRISM_Light light, short rgba[4], PRISM_RenderMode RendMode) {

520void AbstractObject3D::TileRasterisation(int x1, int y1, int x2, int y2, int x3, int y3,
521 float z1, float z2, float z3, SDL_Renderer* SDL_renderer,
522 PRISM_Light light, short rgba[4], PRISM_RenderMode RendMode) {
523 PRISM_Vector3d v0 = {static_cast<float>(x1), static_cast<float>(y1), static_cast<float>(z1)};
524 PRISM_Vector3d v1 = {static_cast<float>(x2), static_cast<float>(y2), static_cast<float>(z2)};
525 PRISM_Vector3d v2 = {static_cast<float>(x3), static_cast<float>(y3), static_cast<float>(z3)};
526
527 // Calculate the normal and other necessary values
528 PRISM_Vector3d normal = Math::CalculateNormals({v0, v1, v2});
529 PRISM_Vector3d viewDir = {0, 0, 1}; // Assuming the camera looks along the negative Z axis
530
531 // Compute bounding box of the triangle
532 float minX = std::min({v0.x, v1.x, v2.x});
533 float maxX = std::max({v0.x, v1.x, v2.x});
534 float minY = std::min({v0.y, v1.y, v2.y});
535 float maxY = std::max({v0.y, v1.y, v2.y});
536
537 // Размер плитки
538 const int tileSize = 32;
539
540 // Обработка каждого блока (плитки)
541 for (int tileY = static_cast<int>(std::floor(minY)) / tileSize;
542 tileY <= static_cast<int>(std::floor(maxY)) / tileSize; ++tileY) {
543 for (int tileX = static_cast<int>(std::floor(minX)) / tileSize;
544 tileX <= static_cast<int>(std::floor(maxX)) / tileSize; ++tileX) {
545
546 // Координаты плитки в пространстве экрана
547 int tileStartX = tileX * tileSize;
548 int tileStartY = tileY * tileSize;
549 int tileEndX = tileStartX + tileSize - 1;
550 int tileEndY = tileStartY + tileSize - 1;
551
552 // Ограничиваем плитку по границам треугольника
553 tileStartX = std::max(tileStartX, static_cast<int>(std::ceil(minX)));
554 tileEndX = std::min(tileEndX, static_cast<int>(std::floor(maxX)));
555 tileStartY = std::max(tileStartY, static_cast<int>(std::ceil(minY)));
556 tileEndY = std::min(tileEndY, static_cast<int>(std::floor(maxY)));
557
558 // Теперь проходим по каждому пикселю внутри плитки
559 for (int y = tileStartY; y <= tileEndY; ++y) {
560 for (int x = tileStartX; x <= tileEndX; ++x) {
561 float u, v, w;
562 if (ComputeBarycentricCoords(static_cast<float>(x), static_cast<float>(y), v0, v1, v2, u, v, w)) {
563 float z = u * v0.z + v * v1.z + w * v2.z;
564 if (z < Camera.z_buffer.get(x, y) || !RendMode.Z_Buffer) {
565 Uint8 r = rgba[0];
566 Uint8 g = rgba[1];
567 Uint8 b = rgba[2];
568 Uint8 a = rgba[3];
569
570 SDL_SetRenderDrawColor(SDL_renderer, r, g, b, a);
571 SDL_RenderDrawPoint(SDL_renderer, x, y);
572 Camera.z_buffer.update(x, y, z);
573 }
574 }
575 }
576 }
577 }
578 }
