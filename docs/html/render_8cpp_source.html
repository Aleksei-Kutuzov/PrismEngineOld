34PRISM_Color PRISM_Render::CalculatePixelSpecular(PRISM_Color originalСolor, PRISM_Light light, AbstractObject3D obj, PRISM_Triangle triangle, PRISM_Vector3d cameraCoordinate)
35{
36
37 PRISM_Vector3d normal = Math::CalculateNormals(triangle);
38 PRISM_Vector3d centroid = { (triangle.a.x + triangle.b.x + triangle.c.x) / 3,
39 (triangle.a.y + triangle.b.y + triangle.c.y) / 3,
40 (triangle.a.z + triangle.b.z + triangle.c.z) / 3 };
41
42 PRISM_Vector3d notNormalLightDir = { light.position.x - centroid.x, light.position.y - centroid.y, light.position.z - centroid.z };
43 PRISM_Vector3d lightDir = Math::Vector_Normalise(notNormalLightDir);
44
45 PRISM_Vector3d reflectDir = (normal * (2.0f * Math::Vector_DotProduct(normal, lightDir))) - lightDir;
46 reflectDir = Math::Vector_Normalise(reflectDir);
47
48 PRISM_Vector3d viewDir = { cameraCoordinate.x - centroid.x, cameraCoordinate.y - centroid.y, cameraCoordinate.z - centroid.z };
49 viewDir = Math::Vector_Normalise(viewDir);
50
51 float spec = std::max(0.f, Math::Vector_DotProduct(reflectDir, viewDir));
52 float shininess = 32.0f;
53 float specular = light.lightIntensity * obj.kSpecular * std::pow(spec, shininess);
54
55 PRISM_Color specularСolor = { originalСolor.r * specular, originalСolor.g * specular, originalСolor.b * specular };
56
57 return specularСolor;
